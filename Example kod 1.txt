scope
	dim a as integer
	dim vysl as integer
	print !"zadejte cislo pro vypocet faktorialu";
	input a
	if a < 0 then
		print !"\nfaktorial nele spocitat\n";
	else
		vysl = 1
		do while A > 0
			vysl = vysl * a
			a = a - 1
		loop
		print !"\nvysledek je:" ; vysl ; !"\n";
	end if
end scope

------------------------------------------------------------------------------
CREATEFRAME (vytvoøení TF)
DEFVAR a
DEFVAR vysl
DEFVAR pomocbool
WRITE !"zadejte cislo pro vypocet faktorialu";
READ TF@a string
LT TF@pomocbool TF@a 0 (true pokud je a < 0)
JUMPIFNEQ else1 lf@pomocbool true ( skoèí na if1 pokud a>0)
write !"\nfaktorial nelze spocitat\n";
JUMP endif

LABEL else1
MOVE TF@vysl 1

LABEL loop
GT TF@pomocbool a 0 (true pro a>0)
JUMPIFEQ loopend lf@pomocbool false (ukonèí loop když a<=0)
MUL TF@vysl TF@vysl TF@a
sub TF@a TF@a 1
JUMP loop

LABEL loopend
WRITE !"\nvysledek je:"
WRITE TF@vysl
WRITE !"\n"

LABEL endif
POPFRAME
